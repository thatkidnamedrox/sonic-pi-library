def nearest?(*args)
  [args].flatten.min_by {|x| (self-x).abs }
end

def pitch
  pitch_class = [:C, :Cs, :D, :Eb, :E, :F, :Fs, :G, :Ab, :A, :Bb, :B]
  pitch_class[self % 12]
end
