def aggregate
  r = self
  r = r.to_a.map {|x| x[1].to_a.zip([x[0]]*x[1].size).to_h }
  w = {}
  r.each do |x|
    w.merge!(x) { |k,v1,v2| ([v1] + [v2]).flatten }
  end
  w
end
